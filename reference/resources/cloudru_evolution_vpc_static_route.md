    }
    virtual_machine = {
      virtual_machine_identifier = {
        id = "8fcc33c4-4580-4146-9b7f-e58943de078e"

      }
    }
    vip = {
      virtual_ip_identifier = [
        {
          id = "8fcc33c4-4580-4146-9b7f-e58943de078e"
        }
